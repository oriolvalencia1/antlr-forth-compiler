>>> from forth import interpret

##########################################
####### Aritmetica
##########################################

>>> interpret("1 2 + .")
'3'

>>> interpret("10 3 - .")
'7'

>>> interpret("6 7 * .")
'42'

>>> interpret("20 4 / .")
'5.0'


##########################################
####### Stack
##########################################

>>> interpret("1 dup .")
'1'

>>> interpret("1 2 swap .")
'1'

>>> interpret("1 2 3 over .")
'2'

>>> interpret("1 2 drop .")
'1'

>>> interpret("10 3 mod .")
'1'


##########################################
####### Boolean
##########################################

>>> interpret("5 3 > .")
'-1'

>>> interpret("3 5 > .")
'0'

>>> interpret("5 5 == .")
'-1'

>>> interpret("-1 not .")
'0'

>>> interpret("0 not .")
'-1'

##########################################
####### Definicions de funcions
##########################################

>>> interpret(": square dup * ; 5 square .")
'25'

>>> interpret(": inc 1 + ; 5 inc .")
'6'


##########################################
####### Condicionals
##########################################

>>> interpret(": abs dup 0 < if 0 swap - endif ; 2 abs .")
'2'

>>> interpret(": abs dup 0 < if 0 swap - endif ; -2 abs .")
'2'

>>> interpret(""" : faux dup rot * swap 1 - dup 2 < if drop else recurse endif ; 
... : factorial dup 2 < if drop 1 else dup 1 - faux endif ;
... 4 factorial .""")
'24'
